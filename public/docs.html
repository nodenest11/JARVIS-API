üö® Error Handling

Understanding JARVIS AI's error handling and response codes.

Error Response Format

{
  "success": false,
  "error": {
    "message": "Human readable error message",
    "code": "ERROR_CODE",
    "timestamp": "2025-07-18T10:30:00.000Z",
    "details": {
      // Additional error context
    }
  }
}

Common Error Codes

Code Description HTTP Status
INVALID_REQUEST Malformed request body 400
NO_PROVIDERS_AVAILABLE All providers are down 503
RATE_LIMIT_EXCEEDED Too many requests 429
PROVIDER_ERROR AI provider returned an error 502

const axios = require('axios');

// Simple chat function
async function chatWithJarvis(message) {
    try {
        const response = await axios.post('/api/chat', {
            message: message
        });
        
        if (response.data.success) {
            console.log('JARVIS:', response.data.data.response);
            console.log('Provider:', response.data.data.provider);
            console.log('Model:', response.data.data.model);
            return response.data.data;
        } else {
            console.error('Error:', response.data.error.message);
            return null;
        }
    } catch (error) {
        console.error('Network error:', error.message);
        return null;
    }
}

            return response.data.data;
        }
        return null;
    } catch (error) {
        console.error('Error checking status:', error.message);
        return null;
    }
}

// Chat function for browser
async function chatWithJarvis(message, resultCallback) {
    try {
        const response = await fetch('/api/chat', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({ message })
        });
        
        const data = await response.json();
        
        if (data.success) {
            resultCallback({
                message: data.data.response,
                provider: data.data.provider,
                model: data.data.model,
                success: true
            });
        } else {
            resultCallback({
                error: data.error.message,
                success: false
            });
        }
    } catch (error) {
        resultCallback({
            error: 'Network error: ' + error.message,
            success: false
        });
    }
}

import requests
import json

class JarvisClient:
    def __init__(self, base_url=""):
        self.base_url = base_url
    
    def chat(self, message):
        """Send a message to JARVIS and get a response"""
        url = f"{self.base_url}/api/chat"
        payload = {"message": message}
        
        try:
            response = requests.post(url, json=payload)
            data = response.json()
            
            if data.get('success'):
                return {
                    'response': data['data']['response'],
                    'provider': data['data']['provider'],
                    'model': data['data']['model'],
                    'success': True
                }
            else:
                return {
                    'error': data['error']['message'],
                    'success': False
                }
                
        except requests.exceptions.RequestException as e:
            return {
                'error': f"Network error: {e}",
                'success': False
            }
    
    def get_status(self):
        """Get system status"""
        url = f"{self.base_url}/api/status"
        
        try:
            response = requests.get(url)
            data = response.json()
            
            if data.get('success'):
                return data['data']
            else:
                return None
                
        except requests.exceptions.RequestException:
            return None
    
    def test_providers(self):
        """Test all AI providers"""
        url = f"{self.base_url}/api/test"

            response = requests.post(url)
            data = response.json()
            
            if data.get('success'):
                return data['data']
            else:
                return None
                
        except requests.exceptions.RequestException:
            return None

<?php
class JarvisClient {
    private $baseUrl;
    
    public function __construct($baseUrl = '') {
        $this->baseUrl = $baseUrl;
    }
    
    public function chat($message) {
        $url = $this->baseUrl . '/api/chat';
        $data = ['message' => $message];
        
        $options = [
            'http' => [
                'method' => 'POST',
                'header' => 'Content-Type: application/json',
                'content' => json_encode($data)
            ]
        ];
        
        $context = stream_context_create($options);
        $result = file_get_contents($url, false, $context);
        
        if ($result === FALSE) {
            return [
                'success' => false,
                'error' => 'Network error'
            ];
        }
        
        $response = json_decode($result, true);
        
        if ($response['success']) {
            return [
                'success' => true,
                'response' => $response['data']['response'],
                'provider' => $response['data']['provider'],
                'model' => $response['data']['model']
            ];
        } else {
            return [
                'success' => false,
                'error' => $response['error']['message']
            ];
        }
    }
    
    public function getStatus() {
        $url = $this->baseUrl . '/api/status';
        $result = file_get_contents($url);
        
        if ($result === FALSE) {
            return null;
        }
        
        $response = json_decode($result, true);
        
        if ($response['success']) {
            return $response['data'];
        }
        
        return null;
    }
}

package main

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io/ioutil"
	"net/http"
)

type ChatRequest struct {
	Message string `json:"message"`
}

type ErrorResponse struct {
	Message string `json:"message"`
}

type ChatResponse struct {
	Success bool `json:"success"`
	Data    struct {
		Response     string `json:"response"`
		Provider     string `json:"provider"`
		Model        string `json:"model"`
		FallbackUsed bool   `json:"fallbackUsed"`
	} `json:"data"`
	Error struct {
		Message string `json:"message"`
	} `json:"error"`
}

func ChatWithJarvis(message string) (*ChatResponse, error) {
	baseURL := "" // Set your base URL here if needed
	url := baseURL + "/api/chat"
	
	requestBody := ChatRequest{
		Message: message,
	}
	
	jsonData, err := json.Marshal(requestBody)
	if err != nil {
		return nil, err
	}
	
	resp, err := http.Post(url, "application/json", bytes.NewBuffer(jsonData))
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	
	body, err := ioutil.ReadAll(resp.Body)
	if err != nil {
		return nil, err
	}
	
	var response ChatResponse
	err = json.Unmarshal(body, &response)
	if err != nil {
		return nil, err
	}
	
	return &response, nil
}

func main() {
	response, err := ChatWithJarvis("Hello JARVIS, how are you today?")
	if err != nil {
		fmt.Printf("Error: %v\n", err)
		return
	}
	
	if response.Success {
		fmt.Printf("JARVIS: %s\n", response.Data.Response)
		fmt.Printf("Provider: %s\n", response.Data.Provider)
		fmt.Printf("Model: %s\n", response.Data.Model)
	} else {
		fmt.Printf("Error: %s\n", response.Error.Message)
	}
}
